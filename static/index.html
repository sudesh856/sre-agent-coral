CORAL SQL QUERIES (live queries powering this investigation)
GitHub — recent pull requests
SELECT number, title, state, user__login, merged_at
FROM github.pulls
WHERE owner = '' AND repo = ''
  AND state = 'closed'
LIMIT 10
Sentry — active errors
SELECT id, title, status, first_seen, level
FROM sentry.issues
ORDER BY first_seen DESC
LIMIT 10
Slack — workspace channels
SELECT id, name, purpose, num_members
FROM slack.channels
ORDER BY num_members DESC
LIMIT 10
Cross-source JOIN — correlate deploys with errors (Coral's core power)
SELECT p.number, p.title, p.merged_at,
       s.title AS error_title, s.first_seen
FROM github.pulls p
JOIN sentry.issues s
  ON s.first_seen <= p.merged_at
WHERE p.owner = '' AND p.repo = ''
ORDER BY p.merged_at DESC
LIMIT 5
